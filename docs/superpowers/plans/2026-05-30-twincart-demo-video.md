# TwinCart Demo Video Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Produce a ≤180s TwinCart demo video, fully automatically: Playwright captures the real running app as 1–2 continuous takes; Remotion composes them with an ElevenLabs voiceover, synced captions, zoom/highlights, and synthetic framing scenes; one command renders `video-out/twincart-demo.mp4`.

**Architecture:** Isolated standalone `remotion/` project (own `node_modules`, not in the pnpm workspace). VO duration drives scene length via a generated `manifest.json` (audio-manifest-driven layout). Twin-finding owns ~55% of runtime; agent checkout is a brief feature. Capture is continuous-take for coherence; Remotion zooms *within* footage by timecode rather than hard-cutting fragments.

**Tech Stack:** Remotion 4 (React 19), @remotion/media-utils, Playwright (Chromium), ElevenLabs `text-to-speech/with-timestamps`, Gemini `gemini-3.1-flash-image-preview`, Bun (scripts), pnpm (the app).

**Spec:** `docs/superpowers/specs/2026-05-30-twincart-demo-video-design.md`

```ts
export type Word = { word: string; start: number; end: number };
export type SceneClip = { scene: string; audioFile: string | null; durationSeconds: number; words: Word[] };
export const MANIFEST = manifest as SceneClip[];
export const FPS = 30;

export const totalFrames = () => MANIFEST.reduce((n, s) => n + sceneFrames(s), 0);
export const sceneStartFrame = (i: number) =>
  MANIFEST.slice(0, i).reduce((n, s) => n + sceneFrames(s), 0);
```

- [ ] **Step 1: Write gen-voiceover.ts**

`scripts/gen-voiceover.ts`:

```ts
import { writeFileSync, mkdirSync } from "node:fs";
import { join } from "node:path";
import { SCENES } from "../remotion/src/data/script";

const API = "https://api.elevenlabs.io/v1/text-to-speech";
const KEY = process.env.ELEVENLABS_API_KEY!;
const VOICE = process.env.ELEVENLABS_VOICE_ID!;
const MODEL = "eleven_multilingual_v2";
const HARD_CAP = 178; // seconds; leaves headroom under the 180s limit
const AUDIO_DIR = join(import.meta.dir, "../remotion/public/audio");
const MANIFEST = join(import.meta.dir, "../remotion/src/data/manifest.json");

type Word = { word: string; start: number; end: number };

// Group ElevenLabs char-level alignment into word-level timings.
function wordsFromAlignment(chars: string[], starts: number[], ends: number[]): Word[] {
  const words: Word[] = [];
  let cur = "", s = 0, e = 0, started = false;
  for (let i = 0; i < chars.length; i++) {
    const c = chars[i];
    if (c === " " || c === "\n") {
      if (cur) { words.push({ word: cur, start: s, end: e }); cur = ""; started = false; }
    } else {
      if (!started) { s = starts[i]; started = true; }
      cur += c; e = ends[i];
    }
  }
  if (cur) words.push({ word: cur, start: s, end: e });
  return words;
}

async function synth(scene: string, text: string) {
  const res = await fetch(`${API}/${VOICE}/with-timestamps`, {
    method: "POST",
    headers: { "xi-api-key": KEY, "content-type": "application/json" },
    body: JSON.stringify({ text, model_id: MODEL }),
  });
  if (!res.ok) throw new Error(`${scene}: ${res.status} ${await res.text()}`);
  const d: any = await res.json();
  const a = d.alignment ?? d.normalized_alignment;
  const ends: number[] = a.character_end_times_seconds;
  const starts: number[] = a.character_start_times_seconds;
  const chars: string[] = a.characters;
  const audioFile = `audio/${scene}.mp3`;
  writeFileSync(join(AUDIO_DIR, `${scene}.mp3`), Buffer.from(d.audio_base64, "base64"));
  return { scene, audioFile, durationSeconds: ends[ends.length - 1], words: wordsFromAlignment(chars, starts, ends) };
}

const main = async () => {
  mkdirSync(AUDIO_DIR, { recursive: true });
  const clips = [];
  for (const s of SCENES) clips.push(await synth(s.id, s.vo));
  const total = clips.reduce((n, c) => n + c.durationSeconds, 0);
  console.table(clips.map((c) => ({ scene: c.scene, sec: c.durationSeconds.toFixed(1) })));
  console.log(`TOTAL ${total.toFixed(1)}s`);
  writeFileSync(MANIFEST, JSON.stringify(clips, null, 2));
  if (total > HARD_CAP) {
    console.error(`OVER CAP: ${total.toFixed(1)}s > ${HARD_CAP}s — trim script.ts and rerun.`);
    process.exit(1);
  }
};
main();
```

```ts
import { chromium } from "playwright";
import { writeFileSync, mkdirSync, renameSync, readdirSync } from "node:fs";
import { join } from "node:path";

const OUT = join(import.meta.dir, "../remotion/public/capture");
const SHOTS = join(import.meta.dir, "../remotion/src/data/shots.ts");
const BASE = process.env.CAPTURE_URL ?? "http://localhost:3000";
const W = 1600, H = 1000; // mobile-first UI reads well here; downscaled in Remotion for zoom headroom
const pause = (ms: number) => new Promise((r) => setTimeout(r, ms));

async function take(name: string, steps: (page: any) => Promise<void>) {
  const browser = await chromium.launch();
  const ctx = await browser.newContext({ viewport: { width: W, height: H }, deviceScaleFactor: 2,
    recordVideo: { dir: OUT, size: { width: W, height: H } } });
  const page = await ctx.newPage();
  await page.goto(BASE, { waitUntil: "networkidle" });
  await steps(page);
  await pause(800);
  await ctx.close(); // finalizes the webm
  await browser.close();
  // rename the autogenerated webm to a stable name
  const webm = readdirSync(OUT).find((f) => f.endsWith(".webm"))!;
  renameSync(join(OUT, webm), join(OUT, `${name}.webm`));
}

const main = async () => {
  mkdirSync(OUT, { recursive: true });

  // TAKE 1 — twin-finding journey (feeds scenes 3 & 4)
  await take("take1", async (page) => {
    const search = page.getByPlaceholder(/Search a product/i);
    await search.click(); await pause(400);
    await search.pressSequentially("robot vacuum", { delay: 90 }); await pause(500);
    await page.getByRole("button", { name: /Find twins/i }).first().click();
    await page.waitForTimeout(1500);                 // results settle
    await page.mouse.wheel(0, 600); await pause(1200); // breathe over results
    await page.mouse.wheel(0, -600); await pause(600);
    await page.getByRole("button", { name: /Compare/i }).first().click(); // -> compare screen
    await page.waitForTimeout(1800);                 // parity view dwell
    await page.mouse.wheel(0, 500); await pause(1500);
    await page.getByRole("button", { name: /Save (savings )?report/i }).first().click();
    await page.waitForTimeout(2200);                 // Box report dwell
  });

  // TAKE 2 — agent checkout (feeds scene 5)
  await take("take2", async (page) => {
    await page.getByPlaceholder(/Search a product/i).fill("robot vacuum");
    await page.getByRole("button", { name: /Find twins/i }).first().click();
    await page.waitForTimeout(1200);
    await page.getByRole("button", { name: /Prepare (agent )?checkout/i }).first().click();
    await page.waitForTimeout(6500);                 // stepper runs its 4 client-side steps
  });

  // shots.ts — approximate timecodes (seconds) per beat within each take.
  // Adjust after watching the takes (Step 3).
  const shots = `// GENERATED by capture-app.ts — timecodes (seconds) into each take.
export const SHOTS = {
  take1: { src: "capture/take1.webm",
    search: [0, 3], results: [3, 9], compare: [9, 16], report: [16, 22] },
  take2: { src: "capture/take2.webm",
    stepper: [2, 9] },
} as const;
`;
  writeFileSync(SHOTS, shots);
  console.log("Captured take1.webm, take2.webm and wrote shots.ts");
};
main();
```

`remotion/src/components/KenBurns.tsx`:

```tsx
import { useCurrentFrame, useVideoConfig, interpolate, AbsoluteFill } from "remotion";

// scaleFrom/scaleTo and focal origin (0..1) drive a smooth push-in.
export const KenBurns: React.FC<{ children: React.ReactNode; from?: number; to?: number; ox?: number; oy?: number }> =
  ({ children, from = 1, to = 1.18, ox = 0.5, oy = 0.5 }) => {
    const f = useCurrentFrame();
    const { durationInFrames } = useVideoConfig();
    const scale = interpolate(f, [0, durationInFrames], [from, to], { extrapolateRight: "clamp" });
```

```ts
import { writeFileSync, mkdirSync } from "node:fs";
import { join } from "node:path";
const DIR = join(import.meta.dir, "../remotion/public/bg");
const KEY = process.env.GEMINI_API_KEY!;
const MODEL = "gemini-3.1-flash-image-preview";
const PROMPTS: Record<string, string> = {
  hook: "Dark cinematic abstract background, deep navy to teal gradient, subtle floating shopping-cart and price-tag silhouettes, premium tech keynote style, no text, 16:9",
  idea: "Two abstract product silhouettes mirrored as twins on a dark teal stage, soft spotlight, minimal, no text, 16:9",
  close: "Dark premium split-stage background, left cool blue right warm green, abstract marketplace nodes, no text, 16:9",
};
async function gen(name: string, prompt: string) {
  const url = `https://generativelanguage.googleapis.com/v1beta/models/${MODEL}:generateContent?key=${KEY}`;
  const r = await fetch(url, { method: "POST", headers: { "content-type": "application/json" },
    body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }] }) });
  const d: any = await r.json();
  const part = d?.candidates?.[0]?.content?.parts?.find((p: any) => p.inlineData);
  if (!part) { console.warn(`no image for ${name}:`, JSON.stringify(d).slice(0, 300)); return; }
  writeFileSync(join(DIR, `${name}.png`), Buffer.from(part.inlineData.data, "base64"));
  console.log("bg", name);
}
const main = async () => { mkdirSync(DIR, { recursive: true });
  for (const [n, p] of Object.entries(PROMPTS)) await gen(n, p); };
main();
```
